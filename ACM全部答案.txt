#include <stdio.h>
int finul(int n);
int main()
{
	int n,m,i,t,max=0,a,b;
	while (scanf("%d %d",&n,&m)!=EOF)
	{
		max=0;
		if(n>m)
		{
			a=n;
			b=m;
			t=n;n=m;m=t;
		}

		for(i=n;i<=m;i++)
		{   
			if(finul(i)>max)
				max=finul(i);
		}
		if(a>b)
		printf("%d %d %d\n",a,b,max+1);
		else 
		printf("%d %d %d\n",n,m,max+1);
	}
	return 0;
}
int finul(int n)
{ 
	int nul=0;
	while(n!=1)
	{
		
		if(n%2==0)
			n/=2;
		     
		else
			n=3*n+1;
		nul++;
	}
	return nul;
}

    scanf("%d",&T);
    while(T--){
        scanf("%d%d%d%d",&A,&B,&n,&m);

            int a,b;

            robt[i].x = a;
            robt[i].y = b;
            if(c == W) robt[i].o = E;
            else if(c == E) robt[i].o = W;
            if(c == N) robt[i].o = S;
            if(c == S) robt[i].o = N;

                }
                g[robt[num].x][robt[num].y] = num;

{ 
    int i; 
    char temp; 
    temp=c[a[0]];

        c[a[i]]=c[a[i+1]];

    int y;

int map[30][30];
void Bfs(int n,int m,Node s,Node t)
{
    queue<Node> q;
    q.push(s);
    while(!q.empty())
    {
        Node node= q.front(); q.pop();
        int x = node.x;
        int y = node.y;
        if(x == t.x && y == t.y)  break;
        for(int i =0; i < 8; i++)
        {
            int cur_x = x + dx[i];
            int cur_y = y + dy[i];
            if(cur_x >= n || cur_x < 0 || cur_y >= m || cur_y < 0)  continue;
            map[cur_x][cur_y] = map[x][y] +1;
            node.x = cur_x;
            node.y = cur_y;
            q.push(node);
        }
    }
}

#include <iostream>

#include <cstdlib>

const int MAXN = 410;

struct bign
{
    int len, s[MAXN];
    bign ()
    {
        memset(s, 0, sizeof(s));
        len = 1;
    }
    bign (int num) { *this = num; }
    bign (const char *num) { *this = num; }
    bign operator = (const int num)
    {
        char s[MAXN];
        sprintf(s, "%d", num);
        *this = s;
        return *this;
    }
    bign operator = (const char *num)
    {
        for(int i = 0; num[i] == '0'; num++) ;  //去前导0
        len = strlen(num);
        for(int i = 0; i < len; i++) s[i] = num[len-i-1] - '0';
        return *this;
    }
    bign operator + (const bign &b) const //+
    {
        bign c;
        c.len = 0;
        for(int i = 0, g = 0; g || i < max(len, b.len); i++)
        {
            int x = g;
            if(i < len) x += s[i];
            if(i < b.len) x += b.s[i];
            c.s[c.len++] = x % 10;
            g = x / 10;
        }
        return c;
    }
    bign operator += (const bign &b)
    {
        *this = *this + b;
        return *this;
    }
    void clean()
    {
        while(len > 1 && !s[len-1]) len--;
    }
    bign operator * (const bign &b) //*
    {
        bign c;
        c.len = len + b.len;
        for(int i = 0; i < len; i++)
        {
            for(int j = 0; j < b.len; j++)
            {
                c.s[i+j] += s[i] * b.s[j];
            }
        }
        for(int i = 0; i < c.len; i++)
        {
            c.s[i+1] += c.s[i]/10;
            c.s[i] %= 10;
        }
        c.clean();
        return c;
    }
    bign operator *= (const bign &b)
    {
        *this = *this * b;
        return *this;
    }
    bign operator - (const bign &b)
    {
        bign c;
        c.len = 0;
        for(int i = 0, g = 0; i < len; i++)
        {
            int x = s[i] - g;
            if(i < b.len) x -= b.s[i];
            if(x >= 0) g = 0;
            else
            {
                g = 1;
                x += 10;
            }
            c.s[c.len++] = x;
        }
        c.clean();
        return c;
    }
    bign operator -= (const bign &b)
    {
        *this = *this - b;
        return *this;
    }
    bign operator / (const bign &b)
    {
        bign c, f = 0;
        for(int i = len-1; i >= 0; i--)
        {
            f = f*10;
            f.s[0] = s[i];
            while(f >= b)
            {
                f -= b;
                c.s[i]++;
            }
        }
        c.len = len;
        c.clean();
        return c;
    }
    bign operator /= (const bign &b)
    {
        *this  = *this / b;
        return *this;
    }
    bign operator % (const bign &b)
    {
        bign r = *this / b;
        r = *this - r*b;
        return r;
    }
    bign operator %= (const bign &b)
